// Web Audio API for "Ding"
const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
function playDing() {
  if(!audioCtx) return;
  const osc = audioCtx.createOscillator();
  const gain = audioCtx.createGain();
  osc.connect(gain); gain.connect(audioCtx.destination);
  osc.type = 'sine'; osc.frequency.setValueAtTime(800, audioCtx.currentTime);
  gain.gain.setValueAtTime(1, audioCtx.currentTime);
  gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.5);
  osc.start(); osc.stop(audioCtx.currentTime + 0.5);
}

  // Try to keep screen awake
  try { if ('wakeLock' in navigator) wakeLock = await navigator.wakeLock.request('screen'); } catch (err) {}

  // UI Changes
  document.getElementById('mainNav').classList.add('hidden');
  document.getElementById('activeWorkoutBar').classList.add('visible');
  document.getElementById('loggerSetup').style.display = 'none';
  document.getElementById('loggerActive').style.display = 'block';
  document.getElementById('train-subtitle').textContent = `In Progress: ${program[dayIdx].dayName}`;

  // Start Global Timer
  clearInterval(globalTimerInt);
  globalTimerInt = setInterval(() => {
    globalSeconds++;
    const m = Math.floor(globalSeconds / 60).toString().padStart(2, '0');
    const s = (globalSeconds % 60).toString().padStart(2, '0');
    document.getElementById('globalWorkoutTimer').textContent = `${m}:${s}`;
  }, 1000);

  renderActiveWorkout();
}

function findGhostData(exName) {
  // Search history backwards for this exercise to use as placeholder
  for (let i = 0; i < history.length; i++) {
    const prevEx = history[i].entries.find(e => e.exercise === exName);
    if (prevEx && prevEx.sets.length > 0) return prevEx.sets;
  }
  return null;
}

function renderActiveWorkout() {
  const day = program[activeDayIdx];
  const el = document.getElementById('loggerContent');
  
  let html = '';
  day.exercises.forEach(ex => {
    const ghostSets = findGhostData(ex.name);
    
    if(!logData[ex.id]) {
      logData[ex.id] = { sets: Array.from({length: ex.sets}, (_, i) => ({ 
        reps: '', weight: '', done: false,
        ghostReps: ghostSets && ghostSets[i] ? ghostSets[i].reps : ex.reps,
        ghostWeight: ghostSets && ghostSets[i] ? ghostSets[i].weight : '-'
      })), timer: 0, notes: '' };
    }
    const entry = logData[ex.id];

    html += `<div class="log-exercise-card" id="card-${ex.id}">
      <div class="log-ex-header">
        <div><div class="log-ex-name">${ex.name}</div><div class="log-ex-target">${ex.sets} Sets</div></div>
        <div class="inline-timer" id="timer-${ex.id}">⏱ Rest</div>
      </div>
      <div class="set-rows">
        ${entry.sets.map((s, si) => `
          <div class="set-row ${s.done?'done':''}" id="setrow-${ex.id}-${si}">
            <div class="set-num">${si+1}</div>
            <div class="set-inputs">
              <div class="set-input-wrap">
                <label>Reps</label>
                <input type="number" placeholder="${s.ghostReps}" value="${s.reps}" inputmode="numeric" onchange="updateSet('${ex.id}',${si},'reps',this.value)">
              </div>
              <div class="set-input-wrap">
                <label>Weight</label>
                <input type="number" placeholder="${s.ghostWeight}" value="${s.weight}" inputmode="decimal" onchange="updateSet('${ex.id}',${si},'weight',this.value)">
              </div>
            </div>
            <div class="set-check" onclick="toggleSet('${ex.id}',${si})"></div>
          </div>`).join('')}
      </div>
    </div>`;
  });
  el.innerHTML = html;
}

function updateSet(exId, si, field, val) { logData[exId].sets[si][field] = val; }

function toggleSet(exId, si) {
  const set = logData[exId].sets[si];
  set.done = !set.done;
  
  // Haptic feedback
  if(set.done && navigator.vibrate) navigator.vibrate(50);

  // Auto-fill from placeholder if empty and checking done
  if (set.done) {
    if (set.reps === '') { set.reps = set.ghostReps; document.querySelector(`#setrow-${exId}-${si} input[type="number"]`).value = set.ghostReps; }
    if (set.weight === '') { set.weight = set.ghostWeight !== '-' ? set.ghostWeight : 0; document.querySelectorAll(`#setrow-${exId}-${si} input`)[1].value = set.weight; }
  }

  const row = document.getElementById(`setrow-${exId}-${si}`);
  row.classList.toggle('done', set.done);

  // Handle Inline Timer
  if(set.done) startInlineTimer(exId);
}

function startInlineTimer(exId) {
  const timerEl = document.getElementById(`timer-${exId}`);
  const cardEl = document.getElementById(`card-${exId}`);
  
  clearInterval(inlineTimers[exId]);
  let timeLeft = REST_TIME;
  
  timerEl.className = 'inline-timer running';
  cardEl.classList.add('active-timer');
  
  inlineTimers[exId] = setInterval(() => {
    timeLeft--;
    const m = Math.floor(timeLeft / 60);
    const s = (timeLeft % 60).toString().padStart(2, '0');
    timerEl.textContent = `${m}:${s}`;
    
    if(timeLeft <= 0) {
      clearInterval(inlineTimers[exId]);
      timerEl.className = 'inline-timer done';
      timerEl.textContent = 'Go!';
      cardEl.classList.remove('active-timer');
      playDing();
      if(navigator.vibrate) navigator.vibrate([200, 100, 200]);
    }
  }, 1000);
}

function finishWorkout() {
  if(!confirm("Complete and save this session?")) return;
  
  // Release WakeLock
  if (wakeLock !== null) { wakeLock.release(); wakeLock = null; }
  
  clearInterval(globalTimerInt);
  Object.values(inlineTimers).forEach(clearInterval);

  const day = program[activeDayIdx];
  const date = new Date().toISOString().split('T')[0]; // YYYY-MM-DD for better sorting
  
  const entries = day.exercises.map(ex => ({
    exercise: ex.name,
    sets: logData[ex.id].sets.filter(s => s.done).map(s => ({ reps: s.reps, weight: s.weight }))
  })).filter(e => e.sets.length > 0);

  if(entries.length > 0) {
    history.unshift({ id: Date.now(), date, dayName: day.dayName, entries });
    localStorage.setItem('ff_history', JSON.stringify(history));
    toast('Session Saved to Analytics ✓', 'success');
  }

  // Reset UI
  isWorkoutActive = false;
  document.getElementById('mainNav').classList.remove('hidden');
  document.getElementById('activeWorkoutBar').classList.remove('visible');
  showPage('history');
}

function exportCSV() {
  if (!history.length) { toast('No data to export.', 'error'); return; }

  // Build Pivot-Table Ready Columns
  let csvContent = "Date,Workout_Day,Exercise,Set_Number,Reps,Weight_Lbs,Est_1RM_Lbs\n";

  history.forEach(session => {
    session.entries.forEach(entry => {
      entry.sets.forEach((set, idx) => {
        const weight = parseFloat(set.weight) || 0;
        const reps = parseInt(set.reps) || 0;
        // Epley 1RM Formula
        const est1RM = weight > 0 && reps > 0 ? (weight * (1 + reps / 30)).toFixed(1) : 0;
        
        // Escape quotes just in case
        const safeDay = `"${session.dayName.replace(/"/g, '""')}"`;
        const safeEx = `"${entry.exercise.replace(/"/g, '""')}"`;
        
        csvContent += `${session.date},${safeDay},${safeEx},${idx + 1},${reps},${weight},${est1RM}\n`;
      });
    });
  });

  // Trigger Download
  const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const link = document.createElement("a");
  link.setAttribute("href", url);
  link.setAttribute("download", `Fortitude_Analytics_${new Date().toISOString().split('T')[0]}.csv`);
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
}
