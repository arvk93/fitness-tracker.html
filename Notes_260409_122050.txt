function showPage(id){
  if(isWorkoutActive && id !== 'logger') { if(!confirm("Workout active. Navigate away?")) return; }
  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n=>n.classList.remove('active'));
  document.getElementById('page-'+id).classList.add('active');
  const nav = document.getElementById('nav-'+id);
  if(nav) nav.classList.add('active');
  window.scrollTo(0,0);
  if(id==='logger' && !isWorkoutActive) setupLoggerPreState();
  if(id==='program') renderDashboard();
}

function toggleAccordion(id, element) {
  const content = document.getElementById(id);
  element.classList.toggle('open');
  content.classList.toggle('open');
}

function advanceWeek() {
  if(!activeProgram) return;
  if(activeProgram.currentWeek >= activeProgram.targetWeeks) {
    toast("You've completed this phase!", "success");
    return;
  }
  if(confirm(`Advance to Week ${activeProgram.currentWeek + 1}?`)) {
    activeProgram.currentWeek += 1;
    saveProgramToStore();
    renderDashboard();
    toast(`Started Week ${activeProgram.currentWeek}`, "success");
  }
}

function renderDashboard() {
  const dashEl = document.getElementById('activeDashboard');
  if(!activeProgram) {
    dashEl.innerHTML = `
      <div class="dashboard-empty">
        <h2 style="color:var(--text)">Ready to train?</h2>
        <p style="color:var(--muted)">You have no active phase. Choose a goal from the Vault below.</p>
      </div>`;
    return;
  }
  
  const currentWeek = activeProgram.currentWeek || 1;
  const pct = (currentWeek / activeProgram.targetWeeks) * 100;
  
  dashEl.innerHTML = `
    <div class="dash-title-row">
      <h2 style="font-size: 1.4rem; font-weight: 800; margin-bottom: 0;">${activeProgram.name}</h2>
      <div class="dash-folder-icon"><svg width="24" height="24" viewBox="0 0 24 24" fill="currentColor"><path d="M22 19a2 2 0 0 1-2 2H4a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h5l2 3h9a2 2 0 0 1 2 2z"></path></svg></div>
    </div>
    
    <div class="tracker-header">Phase Progress</div>
    <div class="progress-bar"><div class="progress-fill" style="width: ${pct}%"></div></div>
    
    <div class="dash-details-grid">
      <div>Week ${currentWeek} of ${activeProgram.targetWeeks}<br>${activeProgram.days.length} workouts/week</div>
      <div>Phase: ${activeProgram.phase.charAt(0).toUpperCase() + activeProgram.phase.slice(1)}<br>${activeProgram.desc || 'Custom block'}</div>
    </div>
    
    <div style="display:flex; flex-direction:column; gap:10px;">
      <button class="btn btn-brand btn-full" onclick="advanceWeek()" ${currentWeek >= activeProgram.targetWeeks ? 'disabled style="opacity:0.5"' : ''}>

      </button>

}

async function startWorkout(dayIdx) {
  activeDayIdx = dayIdx; isWorkoutActive = true; logData = {};
  workoutStartTime = Date.now(); globalSeconds = 0;
  try { if ('wakeLock' in navigator) wakeLock = await navigator.wakeLock.request('screen'); } catch (err) {}
  document.getElementById('mainNavWrap').classList.add('hidden');
  document.getElementById('activeWorkoutBar').classList.add('visible');
  document.getElementById('loggerSetup').style.display = 'none';
  document.getElementById('loggerActive').style.display = 'block';
  document.getElementById('train-header').textContent = activeProgram.days[dayIdx].dayName;
  clearInterval(globalTimerInt); globalTimerInt = setInterval(tickGlobalTimer, 1000);
  renderActiveWorkout(); persistWorkoutState();
}

function tickGlobalTimer(){
  globalSeconds = Math.floor((Date.now() - workoutStartTime) / 1000);
  document.getElementById('globalWorkoutTimer').textContent = `${Math.floor(globalSeconds/60).toString().padStart(2,'0')}:${(globalSeconds%60).toString().padStart(2,'0')}`;
}

function findGhostData(exName) {
  for (let i=0; i<history.length; i++) {
    const prev = history[i].entries.find(e => e.exercise === exName);
    if (prev && prev.sets.length > 0) return { sets: prev.sets, date: history[i].date };
  }
  return null;
}

function renderActiveWorkout() {
  const day = activeProgram.days[activeDayIdx];
  let html = '';
  day.exercises.forEach((ex, exIndex) => {
    const ghost = findGhostData(ex.name);
    const ghostSets = ghost ? ghost.sets : null;
    if(!logData[ex.id]) {
      logData[ex.id] = { sets: Array.from({length: ex.sets}, (_, i) => ({
        reps: '', weight: '', done: false,
        ghostReps: ghostSets && ghostSets[i] ? ghostSets[i].reps : ex.reps,
        ghostWeight: ghostSets && ghostSets[i] ? ghostSets[i].weight : '-'
      }))};
    }
    const entry = logData[ex.id];
    const ytUrl = `https://www.youtube.com/results?search_query=${encodeURIComponent(ex.name + ' exercise tutorial')}`;
    const isBarbell = BARBELL_EXERCISES.has(ex.name);
    const isSuperset = ex.linkedToPrevious;

    html += `<div class="log-exercise-card ${isSuperset ? 'is-superset' : ''}">
      <div class="log-ex-header">
        <div class="log-ex-title-row">
          <div class="log-ex-name">${ex.name}</div>
          <div class="log-ex-actions">
            ${isBarbell ? `<span class="action-link plate" onclick="openPlateCalc()"><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="10"/><circle cx="12" cy="12" r="3"/></svg></span>` : ''}
            <span class="action-link swap" onclick="openSwapModal('${ex.id}', '${ex.name.replace(/'/g, "\\'").replace(/"/g, '&quot;')}')"><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M7 16V4M7 4L3 8M7 4L11 8M17 8V20M17 20L21 16M17 20L13 16"/></svg></span>
            <a href="${ytUrl}" target="_blank" class="action-link yt"><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polygon points="5 3 19 12 5 21 5 3"/></svg></a>
          </div>
        </div>
      </div>
      <div class="set-rows">
        ${entry.sets.map((s, si) => `
          <div class="set-row ${s.done?'done':''}" id="setrow-${ex.id}-${si}">
            <div class="set-num">${si+1}</div>
            <div class="set-inputs">
              <div class="set-input-wrap"><input type="number" placeholder="${s.ghostReps||'-'}" value="${s.reps}" oninput="logData['${ex.id}'].sets[${si}].reps=this.value;persistWorkoutState()"></div>
              <div class="set-input-wrap"><input type="number" placeholder="${s.ghostWeight||'-'}" value="${s.weight}" oninput="logData['${ex.id}'].sets[${si}].weight=this.value;persistWorkoutState()"></div>
            </div>
            <div class="set-check" onclick="toggleSet('${ex.id}', ${si}, ${exIndex})"></div>
          </div>`).join('')}
      </div>
    </div>`;
  });
  document.getElementById('loggerContent').innerHTML = html;
}

function toggleSet(exId, si, exIndex) {
  const set = logData[exId].sets[si]; set.done = !set.done;
  if(set.done && navigator.vibrate) navigator.vibrate(50);
  if (set.done) {
    if (set.reps === '') { set.reps = set.ghostReps || ''; const el = document.querySelector(`#setrow-${exId}-${si} input[type="number"]`); if(el) el.value = set.reps; }
    if (set.weight === '') { set.weight = set.ghostWeight !== '-' ? set.ghostWeight : ''; const els = document.querySelectorAll(`#setrow-${exId}-${si} input`); if(els[1]) els[1].value = set.weight; }
  }
  document.getElementById(`setrow-${exId}-${si}`).classList.toggle('done', set.done);
  persistWorkoutState();

  if(set.done && settings.autoRest) {
    const day = activeProgram.days[activeDayIdx];
    const nextEx = day.exercises[exIndex + 1];
    if (!(nextEx && nextEx.linkedToPrevious)) {
      startTopTimer(settings.defaultRest || 90);
    }
  }
}

    document.getElementById('mainNavWrap').classList.remove('hidden');

async function finishWorkout() {
  if(!confirm("Complete session?")) return;
  if (wakeLock) try { wakeLock.release(); } catch(e){}
  clearInterval(globalTimerInt); 
  const day = activeProgram.days[activeDayIdx];
  const date = getLocalYMD(); 
  
  const entries = day.exercises.map(ex => {
    const validSets = logData[ex.id].sets.filter(s => s.done).map(s => ({ reps: s.reps||0, weight: s.weight||0 }));
    return { exercise: ex.name, sets: validSets };
  }).filter(e => e.sets.length > 0);

  if(entries.length > 0) {
    history.unshift({ id: Date.now(), date, dayName: day.dayName, entries, duration: globalSeconds });
    saveHistoryToStore(); toast('Session Saved ✓', 'success');
  }
  isWorkoutActive = false; localStorage.removeItem('ff_active_workout');
  document.getElementById('mainNavWrap').classList.remove('hidden');
  document.getElementById('activeWorkoutBar').classList.remove('visible');
  showPage('program');
}

function startTopTimer(seconds) {
  ensureAudio(); clearInterval(topTimerInt); topTimerRemaining = seconds;
  const disp = document.getElementById('topTimerDisplay');
  disp.classList.remove('finished'); disp.classList.add('running');
  updateTopTimerText();
  topTimerInt = setInterval(() => {
    topTimerRemaining--; updateTopTimerText();
    if(topTimerRemaining <= 0) {
      clearInterval(topTimerInt);
      disp.classList.remove('running'); disp.classList.add('finished');
      playDing();
      if(settings.sound && navigator.vibrate) navigator.vibrate([250, 100, 250, 100, 400]);
      setTimeout(() => { disp.classList.remove('finished'); disp.textContent = "00:00"; }, 3500);
    }
  }, 1000);
}
function stopTopTimer() { clearInterval(topTimerInt); const disp = document.getElementById('topTimerDisplay'); disp.classList.remove('running','finished'); disp.textContent = "00:00"; }

function confirmSwap(newName) {
  const ex = activeProgram.days[activeDayIdx].exercises.find(e => e.id === currentSwapExId);
  if(ex) {
    ex.name = newName;
    const ghost = findGhostData(newName); const ghostSets = ghost ? ghost.sets : null;
    logData[currentSwapExId].sets.forEach((s, i) => {
      s.reps = ''; s.weight = ''; s.done = false;
      s.ghostReps = ghostSets && ghostSets[i] ? ghostSets[i].reps : ex.reps;
      s.ghostWeight = ghostSets && ghostSets[i] ? ghostSets[i].weight : '-';
    });
    renderActiveWorkout(); persistWorkoutState(); toast('Swapped', 'success');
  }
  closeModal('swapModal');
}

const PLATE_SIZES = [
  { lbs: 45, color: '#00D8FF', label: '45' },
  { lbs: 35, color: '#FFD60A', label: '35' },
  { lbs: 25, color: '#32D74B', label: '25' },
  { lbs: 10, color: 'rgba(255,255,255,0.2)', label: '10' },
  { lbs: 5, color: '#FF453A', label: '5' },
  { lbs: 2.5, color: 'rgba(255,255,255,0.1)', label: '2.5' },
];
function openPlateCalc() { document.getElementById('plateTarget').value = ''; renderPlateCalc(); document.getElementById('plateModal').classList.add('open'); }
function renderPlateCalc() {
  const target = parseFloat(document.getElementById('plateTarget').value) || 0;
  const bar = parseFloat(document.getElementById('plateBar').value) || 45;
  const totalEl = document.getElementById('plateTotal');
  const subEl = document.getElementById('plateSub');
  const breakdownEl = document.getElementById('plateBreakdown');

  if(!target || target < bar) { totalEl.textContent = target ? target.toFixed(0) + ' lb' : '—'; subEl.textContent = target ? `Below bar weight (${bar} lb)` : 'Enter a target weight above'; breakdownEl.innerHTML = ''; return; }
  const perSide = (target - bar) / 2; let remaining = perSide; const plates = [];
  PLATE_SIZES.forEach(p => { const count = Math.floor(remaining / p.lbs); if(count > 0) { plates.push({ ...p, count }); remaining -= count * p.lbs; } });
  totalEl.textContent = target.toFixed(0) + ' lb';
  const loaded = bar + 2 * (perSide - remaining);
  subEl.textContent = remaining > 0.01 ? `Closest: ${loaded.toFixed(1)} lb (${remaining.toFixed(2)} lb short per side)` : `Per side: ${perSide.toFixed(1)} lb`;
  if(plates.length === 0) { breakdownEl.innerHTML = '<p style="text-align:center; color:var(--muted)">Just the bar</p>'; } 
  else { breakdownEl.innerHTML = plates.map(p => `<div class="plate-row"><div class="plate-dot" style="background:${p.color}">${p.label}</div><div class="plate-label">${p.lbs} lb plate</div><div class="plate-count">${p.count}×</div></div>`).join(''); }
}
